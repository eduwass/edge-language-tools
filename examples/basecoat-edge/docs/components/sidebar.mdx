Application sidebar navigation; requires basecoat sidebar JS (all.min.js).

## Props

```ts
{
  id?: string
  label?: string
  open?: boolean
  side?: 'left' | 'right'
  menu?: Array<{
    type?: 'item' | 'group' | 'separator' | 'submenu'
    label?: string
    url?: string
    icon?: string
    current?: boolean
    open?: boolean
    items?: Array<{
      type?: 'item' | 'group' | 'separator' | 'submenu'
      label?: string
      url?: string
      icon?: string
      current?: boolean
      open?: boolean
      items?: unknown[]
    }>
    attrs?: Record<string, string>
    id?: string
  }>
  main_attrs?: Record<string, string>
  [attr: string]: unknown
}
```

## Usage

```edge
@@sidebar({ /* see props */ })
  Content
@end
```
