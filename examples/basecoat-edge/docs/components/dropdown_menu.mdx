Dropdown menu with trigger and items; requires basecoat dropdown-menu JS (all.min.js).

## Props

```ts
{
  id: string
  items?: Array<{
    type?: 'item' | 'group' | 'separator'
    label?: string
    url?: string
    icon?: string
    items?: Array<{
      type?: 'item' | 'group' | 'separator'
      label?: string
      url?: string
      icon?: string
      attrs?: Record<string, string>
    }>
    attrs?: Record<string, string>
    id?: string
  }>
  trigger_attrs?: Record<string, string>
  main_attrs?: Record<string, string>
  [attr: string]: unknown
}
```

## Usage

```edge
@@dropdownMenu({ /* required props */ })
  Content
@end
```
